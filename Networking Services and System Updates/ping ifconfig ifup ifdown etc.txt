[root@mylinuxvm network-scripts]# cd /etc/sysconfig/network-scripts/

ls  -ltr --> in the bottom you will have ifcfg-enp0s3

[root@mylinuxvm network-scripts]# vi ifcfg-enp0s3
TYPE=Ethernet
PROXY_METHOD=none
BROWSER_ONLY=no
BOOTPROTO=dhcp  --> this where you change your ip to static. so every times you 
		    reboot your ip address does not change this way.
DEFROUTE=yes
IPV4_FAILURE_FATAL=no
IPV6INIT=yes
IPV6_AUTOCONF=yes
IPV6_DEFROUTE=yes
IPV6_FAILURE_FATAL=no
IPV6_ADDR_GEN_MODE=stable-privacy
NAME=enp0s3
UUID=cd3ff13b-02f0-43f2-beac-9aaf5a19b1fc
DEVICE=enp0s3
ONBOOT=yes
IPV6_PRIVACY=no
